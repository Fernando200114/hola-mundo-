if ingreso == ingr:                                             #Colocamos la funcion if donde el ingreso va a comparar ingeso con ingr
  print("es un polidromo")                                      #En el cual si se cumple sera un polidromo caso contrario no lo sera
else:
  print("no es un polidromo")
